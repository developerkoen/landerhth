const CORRECT_USERNAME = "admin"; const CORRECT_PASSWORD = "1234"; function login() { let username = document.getElementById("username").value; let password = document.getElementById("password").value; let message = document.getElementById("message"); if (username === CORRECT_USERNAME && password === CORRECT_PASSWORD) { message.style.color = "green"; message.textContent = "Succesvol ingelogd!"; // Sla in localStorage op dat de gebruiker is ingelogd localStorage.setItem("loggedIn", "true"); // Wacht 1,5 sec en stuur terug naar index.html setTimeout(() => { window.location.href = "index.html"; }, 1500); } else { message.style.color = "red"; message.textContent = "Onjuiste inloggegevens!"; } } // Controleer of gebruiker al is ingelogd bij het laden van index.html window.onload = function() { if (localStorage.getItem("loggedIn") === "true") { document.getElementById("message").textContent = "Je bent al ingelogd!"; document.getElementById("message").style.color = "green"; } };
